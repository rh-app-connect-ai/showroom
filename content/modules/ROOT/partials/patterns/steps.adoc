<script>
window.addEventListener('DOMContentLoaded', function() {

  var overlay = document.getElementById('steps-overlay');
  if (!overlay) return;

  var cards = document.querySelectorAll('.tile-card[data-module]');
  if (!cards.length) return;

  cards.forEach(function(card) {
    var prefix = card.getAttribute('data-module');
    if (!prefix) return;

    // Skip cards that have a resume URL — navigate directly
    var resumeUrl = localStorage.getItem('resume:' + prefix);
    if (resumeUrl) return;

    var anchor = card.querySelector('a');
    if (!anchor) return;

    anchor.addEventListener('click', function(e) {
      // Re-check resume at click time (may have changed since page load)
      if (localStorage.getItem('resume:' + prefix)) return;

      e.preventDefault();

      // Populate overlay
      var moduleName = card.querySelector('.tile-header').textContent;
      document.getElementById('steps-module').textContent = moduleName;

      // Get first link (original, before resume override)
      var firstLink = card.getAttribute('data-first-link');
      document.getElementById('steps-start').setAttribute('href', firstLink);

      // Collect page titles from all nav links matching the module prefix
      // When the same href appears twice (parent * and child **), use the child's text
      var pageTitles = [];
      var seen = {};
      var allNavLinks = document.querySelectorAll('a.nav-link');
      for (var i = 0; i < allNavLinks.length; i++) {
        var href = allNavLinks[i].getAttribute('href');
        if (href && href.indexOf(prefix) !== -1) {
          if (seen[href] !== undefined) {
            pageTitles[seen[href]] = allNavLinks[i].textContent.trim();
          } else {
            seen[href] = pageTitles.length;
            pageTitles.push(allNavLinks[i].textContent.trim());
          }
        }
      }

      // Get per-page times from data-times (set by tiles.adoc fetch)
      var timesJson = card.getAttribute('data-times');
      var times = timesJson ? JSON.parse(timesJson) : [];

      // Build steps list
      var listEl = document.getElementById('steps-list');
      listEl.innerHTML = '';
      var totalTime = 0;
      for (var k = 0; k < pageTitles.length; k++) {
        var t = times[k] || 1;
        totalTime += t;
        var item = document.createElement('div');
        item.className = 'steps-item';
        item.innerHTML =
          '<span class="steps-item-title">' + pageTitles[k] + '</span>' +
          '<span class="steps-item-time">' + t + ' min</span>';
        listEl.appendChild(item);
      }

      // Update total time
      document.getElementById('steps-total').textContent = totalTime + ' min total';

      // Show overlay
      overlay.style.display = 'flex';
    });
  });

  // Close overlay: X button
  document.getElementById('steps-close').addEventListener('click', function() {
    overlay.style.display = 'none';
  });

  // Close overlay: click on background (outside content)
  overlay.addEventListener('click', function(e) {
    if (e.target === overlay) {
      overlay.style.display = 'none';
    }
  });

});
</script>
