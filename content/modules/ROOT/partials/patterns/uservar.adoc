<script>
window.addEventListener('DOMContentLoaded', function() {

  var source = document.getElementById('uv-source');
  if (!source) return;

  // Check if uv-source contains a .uservar span (production mode)
  // or plain text (local dev mode)
  var innerSpan = source.querySelector('.uservar');

  if (!innerSpan) {
    // Local dev: attribute resolved at build time, set the cookie
    var username = source.textContent.trim();
    if (username) {
      document.cookie = 'username=' + encodeURIComponent(username) + '; Path=/';
    }
    return;
  }

  // Production: read cookies and replace .uservar spans
  var cookies = {};
  document.cookie.split(';').forEach(function(c) {
    var parts = c.trim().split('=');
    if (parts.length === 2) {
      cookies[parts[0].trim()] = decodeURIComponent(parts[1].trim());
    }
  });

  var spans = document.querySelectorAll('.uservar');
  spans.forEach(function(span) {
    var key = span.textContent.trim();
    if (cookies[key] !== undefined) {
      span.textContent = cookies[key];
    }
  });

  // Logout: redirect to docs-proxy /logout endpoint, which clears
  // the httponly _oauth_proxy and username cookies server-side,
  // then redirects back to docs-proxy to trigger a fresh OAuth flow.
  var host = window.location.hostname;
  var appsIndex = host.indexOf('.apps.');
  if (appsIndex !== -1) {
    var domain = host.substring(appsIndex + 1);
    var logoutUrl = 'https://docs-showroom.' + domain + '/logout';
    var logoutLinks = document.querySelectorAll('.navbar-item-right-content');
    logoutLinks.forEach(function(link) {
      if (link.textContent.trim() === 'Log out') {
        link.href = logoutUrl;
        link.target = '_self';
      }
    });
  }

});
</script>
