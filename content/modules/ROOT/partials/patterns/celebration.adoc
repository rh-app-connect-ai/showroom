document.addEventListener('DOMContentLoaded', function() {

  // Find Antora's Next page link
  var navNext = document.querySelector('.nav-next a')
    || document.querySelector('.pagination .next a')
    || document.querySelector('a.page-next');
  if (!navNext) {
    document.querySelectorAll('a').forEach(function(a) {
      if (a.closest('.toolbar') || a.closest('.nav-container')) {
        if (a.textContent.trim().toLowerCase().indexOf('next') !== -1) navNext = a;
      }
    });
  }
  if (!navNext) return;

  // Only activate on the last page of a module:
  // if the Next link points to the same directory, this is a mid-module page
  var currentDir = window.location.pathname.replace(/[^\/]*$/, '');
  var nextUrl = new URL(navNext.getAttribute('href'), window.location.href);
  var nextDir = nextUrl.pathname.replace(/[^\/]*$/, '');
  if (currentDir === nextDir) return;

  // Replace Next link text with "Finish"
  navNext.textContent = 'Finish';

  // Find module name from nav sidebar
  var moduleName = '';
  var currentItem = document.querySelector('.nav-item.is-current-page');
  if (currentItem) {
    var el = currentItem;
    while (el) {
      var parent = el.parentElement ? el.parentElement.closest('.nav-item') : null;
      if (!parent) break;
      el = parent;
    }
    var link = el.querySelector(':scope > a.nav-link');
    if (link) moduleName = link.textContent.trim();
  }

  // Find home URL from nav
  var homeUrl = './';
  var homeLink = document.querySelector('.nav-list > .nav-item:first-child > a.nav-link');
  if (homeLink) homeUrl = homeLink.getAttribute('href');

  // Determine module prefix from current URL
  var pathMatch = window.location.pathname.match(/\/([^\/]+\/)[^\/]+?(?:\.html)?$/);
  var modulePrefix = pathMatch ? pathMatch[1] : '';

  // Module progress check â€” same algorithm as tiles.js
  function getModuleProgress() {
    var totalTasks = parseInt(localStorage.getItem('module:' + modulePrefix + 'tasks'), 10);
    if (!totalTasks) return -1;

    var pagesWithChecks = {};
    var visitedPages = {};
    for (var i = 0; i < localStorage.length; i++) {
      var key = localStorage.key(i);
      if (key.indexOf('verify:' + modulePrefix) === 0 && key.endsWith(':total')) {
        var pageKey = key.slice(7, -6);
        var pageTotal = parseInt(localStorage.getItem(key), 10) || 0;
        var pageCompleted = 0;
        for (var j = 0; j < pageTotal; j++) {
          if (localStorage.getItem('verify:' + pageKey + ':' + j) === 'yes') pageCompleted++;
        }
        pagesWithChecks[pageKey] = { total: pageTotal, completed: pageCompleted };
      }
      if (key.indexOf('visited:' + modulePrefix) === 0) {
        visitedPages[key.slice(8)] = true;
      }
    }
    var earned = 0;
    for (var pk in pagesWithChecks) {
      earned += pagesWithChecks[pk].completed / pagesWithChecks[pk].total;
    }
    for (var vk in visitedPages) {
      if (!pagesWithChecks[vk]) earned += 1;
    }
    return Math.min(100, Math.round((earned / totalTasks) * 100));
  }

  // Keep Next link disabled until module progress is 100%
  function updateNextState() {
    var progress = getModuleProgress();
    if (progress >= 0 && progress < 100) {
      navNext.style.pointerEvents = 'none';
      navNext.style.opacity = '0.3';
    } else {
      navNext.style.pointerEvents = '';
      navNext.style.opacity = '';
    }
  }

  document.addEventListener('change', function(e) {
    if (e.target.type === 'radio') setTimeout(updateNextState, 0);
  });

  updateNextState();

  // Populate dynamic content
  var overlay = document.getElementById('celebration-overlay');
  var moduleEl = document.getElementById('celebration-module');
  moduleEl.innerHTML = moduleName
    ? 'You completed <strong>' + moduleName + '</strong>'
    : 'You completed this module';
  document.getElementById('celebration-home').href = homeUrl;

  // Intercept Next link click â†’ show celebration instead
  navNext.addEventListener('click', function(e) {
    e.preventDefault();
    overlay.style.display = 'flex';
  });

});
</script>
