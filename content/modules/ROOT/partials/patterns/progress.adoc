<script>
document.addEventListener('DOMContentLoaded', function() {
  var pathMatch = window.location.pathname.match(/\/([^\/]+\/[^\/]+?)(?:\.html)?$/);
  if (!pathMatch) return;
  var pageKey = pathMatch[1];
  localStorage.setItem('visited:' + pageKey, 'true');
  // Save last visited URL per module for resume
  var prefixMatch = pageKey.match(/([^\/]*\/)/);
  if (prefixMatch) localStorage.setItem('resume:' + prefixMatch[1], window.location.pathname);

  // Override "Previous" link when it crosses module boundaries → go to dashboard
  var currentPrefix = prefixMatch ? prefixMatch[1] : '';
  if (currentPrefix) {
    var prevLink = document.querySelector('.nav-prev a')
      || document.querySelector('.pagination .prev a');
    if (prevLink) {
      var prevHref = prevLink.getAttribute('href');
      if (prevHref && prevHref.indexOf(currentPrefix) === -1) {
        var homeLink = document.querySelector('.nav-list > .nav-item:first-child > a.nav-link');
        prevLink.setAttribute('href', homeLink ? homeLink.getAttribute('href') : './');
        prevLink.textContent = 'Dashboard';
      }
    }
  }
});
</script>
