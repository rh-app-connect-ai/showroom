<script>
window.addEventListener('DOMContentLoaded', function() {

  // Transform AsciiDoc .verification sidebar blocks into verification-box HTML
  var sidebars = document.querySelectorAll('.sidebarblock.verification');
  sidebars.forEach(function(sb, idx) {
    var content = sb.querySelector('.content');
    if (!content) return;
    var paragraphs = content.querySelectorAll('.paragraph');
    var question = '', successMsg = '', failMsg = '';
    paragraphs.forEach(function(p) {
      if (p.classList.contains('success')) {
        successMsg = p.querySelector('p') ? p.querySelector('p').innerHTML : p.textContent;
      } else if (p.classList.contains('fail')) {
        failMsg = p.querySelector('p') ? p.querySelector('p').innerHTML : p.textContent;
      } else if (!question) {
        question = p.querySelector('p') ? p.querySelector('p').innerHTML : p.textContent;
      }
    });
    if (!successMsg) successMsg = 'Great, you may continue to the next section.';
    if (!failMsg) failMsg = 'Review your steps and try again.';
    var name = 'verify-' + idx;
    var box = document.createElement('div');
    box.className = 'verification-box';
    box.innerHTML =
      '<div class="vb-header">Check your work</div>' +
      '<div class="vb-body">' +
        '<div class="vb-question">' + question + '</div>' +
        '<div class="vb-options">' +
          '<label><input type="radio" name="' + name + '" value="yes"> Yes</label>' +
          '<label><input type="radio" name="' + name + '" value="no"> No</label>' +
        '</div>' +
        '<div class="vb-feedback success">' + successMsg + '</div>' +
        '<div class="vb-feedback fail">' + failMsg + '</div>' +
      '</div>';
    sb.parentNode.replaceChild(box, sb);
  });

  // Now process all verification boxes (both pre-existing HTML and transformed ones)
  var boxes = document.querySelectorAll('.verification-box');
  if (!boxes.length) return;

  // Find Antora's Next page link
  var navNext = document.querySelector('.nav-next a')
    || document.querySelector('.pagination .next a')
    || document.querySelector('a.page-next');
  if (!navNext) {
    document.querySelectorAll('a').forEach(function(a) {
      if (a.closest('.toolbar') || a.closest('.nav-container')) {
        if (a.textContent.trim().toLowerCase().indexOf('next') !== -1) navNext = a;
      }
    });
  }

  // Disable Next link on load
  if (navNext) {
    navNext.style.pointerEvents = 'none';
    navNext.style.opacity = '0.3';
  }

  function checkAll() {
    var allYes = true;
    boxes.forEach(function(box) {
      var selected = box.querySelector('input[type="radio"]:checked');
      if (!selected || selected.value !== 'yes') allYes = false;
    });
    if (navNext) {
      navNext.style.pointerEvents = allYes ? '' : 'none';
      navNext.style.opacity = allYes ? '' : '0.3';
    }
  }

  boxes.forEach(function(box) {
    box.querySelectorAll('input[type="radio"]').forEach(function(r) {
      r.addEventListener('change', function() {
        var ok = box.querySelector('.vb-feedback.success');
        var fail = box.querySelector('.vb-feedback.fail');
        if (this.value === 'yes') {
          ok.style.display = 'block'; fail.style.display = 'none';
        } else {
          ok.style.display = 'none'; fail.style.display = 'block';
        }
        checkAll();
      });
    });
  });

});
</script>
