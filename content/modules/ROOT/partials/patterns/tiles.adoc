<script>
window.addEventListener('DOMContentLoaded', function() {

  // Transform AsciiDoc .tile sidebar blocks into tile-card HTML
  var tiles = document.querySelectorAll('.sidebarblock.tile');
  if (!tiles.length) return;

  var grid = document.createElement('div');
  grid.className = 'tile-grid';
  var clockSvg = '<svg width="14" height="14" viewBox="0 0 24 24" fill="none" stroke="#777" stroke-width="2"><circle cx="12" cy="12" r="10"/><polyline points="12 6 12 12 16 14"/></svg>';
  tiles[0].parentNode.insertBefore(grid, tiles[0]);
  tiles.forEach(function(sb) {
    var content = sb.querySelector('.content');
    if (!content) return;
    var paragraphs = content.querySelectorAll('.paragraph');
    var link = '', title = '', desc = '';
    paragraphs.forEach(function(p) {
      var a = p.querySelector('a');
      if (a && !link) {
        link = a.getAttribute('href');
        title = a.textContent;
      } else if (p.classList.contains('meta')) {
        // Ignored — time/tasks are now dynamic
      } else if (link) {
        desc += (desc ? ' ' : '') + (p.querySelector('p') ? p.querySelector('p').textContent : p.textContent).trim();
      }
    });
    // Extract module prefix from xref link (e.g. "m1/m1.0.html" → "m1/")
    var prefixMatch = link.match(/([^\/]*\/)/);
    var modulePrefix = prefixMatch ? prefixMatch[1] : '';

    var card = document.createElement('div');
    card.className = 'tile-card';
    card.setAttribute('data-module', modulePrefix);
    card.setAttribute('data-first-link', link);
    card.innerHTML =
      '<a href="' + link + '">' +
        '<div class="tile-header">' + title + '</div>' +
        '<div class="tile-body">' +
          '<div class="tile-desc">' + desc + '</div>' +
          '<div class="tile-footer">' +
            '<span class="tile-time"></span>' +
            '<span class="tile-tasks"></span>' +
          '</div>' +
          '<div class="tile-progress" style="display:none"></div>' +
        '</div>' +
      '</a>';

    // Resume: link to last visited page instead of overview
    var resumeUrl = localStorage.getItem('resume:' + modulePrefix);
    if (resumeUrl) card.querySelector('a').setAttribute('href', resumeUrl);

    grid.appendChild(card);
    sb.remove();
  });

  // Populate task counts and times dynamically from the nav sidebar
  var cards = grid.querySelectorAll('.tile-card[data-module]');
  cards.forEach(function(card) {
    var prefix = card.getAttribute('data-module');
    if (!prefix) return;

    // Find all nav links whose href contains the module prefix
    var pageUrls = [];
    var seen = {};
    var allNavLinks = document.querySelectorAll('a.nav-link');
    for (var i = 0; i < allNavLinks.length; i++) {
      var href = allNavLinks[i].getAttribute('href');
      if (href && href.indexOf(prefix) !== -1 && !seen[href]) {
        seen[href] = true;
        pageUrls.push(href);
      }
    }

    var tasksCount = pageUrls.length;
    if (!tasksCount) return;

    // Show task count immediately
    card.setAttribute('data-tasks', tasksCount);
    card.querySelector('.tile-tasks').textContent = tasksCount + ' tasks';
    localStorage.setItem('module:' + prefix + 'tasks', tasksCount);

    // Fetch pages in background to get per-page times
    Promise.all(pageUrls.map(function(url) {
      return fetch(url).then(function(r) { return r.text(); }).catch(function() { return ''; });
    })).then(function(htmls) {
      var times = htmls.map(function(html) {
        if (!html) return 1;
        var doc = new DOMParser().parseFromString(html, 'text/html');
        var span = doc.querySelector('.page-time');
        return span ? parseInt(span.getAttribute('data-time'), 10) || 1 : 1;
      });
      var total = times.reduce(function(a, b) { return a + b; }, 0);
      card.querySelector('.tile-time').innerHTML = clockSvg + ' ' + total + ' min';
      card.setAttribute('data-times', JSON.stringify(times));
    });
  });

  // Build progress bars from localStorage
  // Each page (task) has equal weight = 1. Pages with checks split their
  // weight across the checks: each check is worth 1/numChecks.
  // Pages without checks earn full credit when visited.
  cards.forEach(function(card) {
    var prefix = card.getAttribute('data-module');
    var totalTasks = parseInt(card.getAttribute('data-tasks'), 10);
    if (!prefix || !totalTasks) return;

    // Collect page data from localStorage
    var pagesWithChecks = {};  // pageKey → { total, completed }
    var visitedPages = {};     // pageKey → true

    for (var i = 0; i < localStorage.length; i++) {
      var key = localStorage.key(i);

      // Verification totals: "verify:m1/m1.2:total"
      if (key.indexOf('verify:' + prefix) === 0 && key.endsWith(':total')) {
        var pageKey = key.slice(7, -6); // strip "verify:" and ":total"
        var pageTotal = parseInt(localStorage.getItem(key), 10) || 0;
        var pageCompleted = 0;
        for (var j = 0; j < pageTotal; j++) {
          if (localStorage.getItem('verify:' + pageKey + ':' + j) === 'yes') pageCompleted++;
        }
        pagesWithChecks[pageKey] = { total: pageTotal, completed: pageCompleted };
      }

      // Page visits: "visited:m1/m1.2"
      if (key.indexOf('visited:' + prefix) === 0) {
        visitedPages[key.slice(8)] = true; // strip "visited:"
      }
    }

    // Calculate earned weight
    var earned = 0;
    // Pages with checks: each page worth 1, split across its checks
    for (var pk in pagesWithChecks) {
      var info = pagesWithChecks[pk];
      earned += info.completed / info.total;
    }
    // Visited pages without checks: each worth 1
    for (var vk in visitedPages) {
      if (!pagesWithChecks[vk]) earned += 1;
    }

    if (earned === 0) return;
    var pct = Math.min(100, Math.round((earned / totalTasks) * 100));
    var progDiv = card.querySelector('.tile-progress');
    progDiv.style.display = '';
    if (pct === 100) {
      card.querySelector('.tile-header').style.backgroundColor = '#3e8635';
    }
    progDiv.innerHTML =
      '<div class="tile-progress-track">' +
        '<div class="tile-progress-fill" style="width:' + pct + '%"></div>' +
      '</div>' +
      '<div class="tile-progress-info">' +
        (pct === 100 ? '<span class="tile-completed-label">\u2705 Completed</span>' : '<span></span>') +
        '<span class="tile-progress-label">' + pct + '%</span>' +
      '</div>';
  });

  // Reset progress button
  var reset = document.createElement('div');
  reset.style.cssText = 'text-align:right; margin-top:8px; font-size:12px;';
  reset.innerHTML = '<a href="#" style="color:#999;">Reset progress</a>';
  reset.querySelector('a').addEventListener('click', function(e) {
    e.preventDefault();
    Object.keys(localStorage).filter(function(k) {
      return k.indexOf('verify:') === 0 || k.indexOf('visited:') === 0 || k.indexOf('resume:') === 0 || k.indexOf('module:') === 0;
    }).forEach(function(k) { localStorage.removeItem(k); });
    location.reload();
  });
  grid.parentNode.insertBefore(reset, grid.nextSibling);

});
</script>
