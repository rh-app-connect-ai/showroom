  border: 1px solid #ddd;

  overflow: hidden;

  display: flex;

.tile-card a {
  text-decoration: none;
  color: inherit;
  display: flex;
  flex-direction: column;
  height: 100%;
}
.tile-card .tile-header {
  background-color: #3e8635;

<script>
window.addEventListener('DOMContentLoaded', function() {

  // Transform AsciiDoc .tile sidebar blocks into tile-card HTML
  var tiles = document.querySelectorAll('.sidebarblock.tile');
  if (!tiles.length) return;

  var grid = document.createElement('div');
  grid.className = 'tile-grid';
  var clockSvg = '<svg width="14" height="14" viewBox="0 0 24 24" fill="none" stroke="#777" stroke-width="2"><circle cx="12" cy="12" r="10"/><polyline points="12 6 12 12 16 14"/></svg>';
  tiles[0].parentNode.insertBefore(grid, tiles[0]);
  tiles.forEach(function(sb) {
    var content = sb.querySelector('.content');
    if (!content) return;
    var paragraphs = content.querySelectorAll('.paragraph');
    var link = '', title = '', desc = '', time = '', tasks = '';
    paragraphs.forEach(function(p) {
      var a = p.querySelector('a');
      if (a && !link) {
        link = a.getAttribute('href');
        title = a.textContent;
      } else if (p.classList.contains('meta')) {
        var text = p.textContent.trim();
        var parts = text.split('|');
        if (parts.length >= 1) time = parts[0].trim();
        if (parts.length >= 2) tasks = parts[1].trim();
      } else if (link) {
        desc += (desc ? ' ' : '') + (p.querySelector('p') ? p.querySelector('p').textContent : p.textContent).trim();
      }
    });
    var card = document.createElement('div');
    card.className = 'tile-card';
    card.innerHTML =
      '<a href="' + link + '">' +
        '<div class="tile-header">' + title + '</div>' +
        '<div class="tile-body">' +
          '<div class="tile-desc">' + desc + '</div>' +
          (time || tasks ?
            '<div class="tile-footer">' +
              (time ? '<span class="tile-time">' + clockSvg + ' ' + time + '</span>' : '') +
              (tasks ? '<span class="tile-tasks">' + tasks + '</span>' : '') +
            '</div>' : '') +
        '</div>' +
      '</a>';
    grid.appendChild(card);
    sb.remove();
  });

});
</script>
