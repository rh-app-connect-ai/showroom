. Configure the AI model
+
Open `application.properties` and add the following configuration for the AI agent:
+
{blank}
+
[.copypaste]
****
[.path]
/credentials/model

[.default]
https://model-url, model-api-key

# Model configuration
forage.myagent.agent.model.kind=openai
forage.myagent.agent.model.name=qwen3-14b
forage.myagent.agent.base.url=REPLACE
forage.myagent.agent.api.key=REPLACE

# Memory settings
forage.myagent.agent.features=memory
forage.myagent.agent.memory.kind=message-window
forage.myagent.agent.memory.max.messages=20

# Log configuration
forage.myagent.agent.log.requests=true
forage.myagent.agent.log.responses=true
